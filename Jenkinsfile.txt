pipeline {
    agent any
    

    
    stages {
        stage('1-Build') {
            steps {
                sh "pip3 install -r requirements.txt"
                echo "Running pip3"
            }
        }
        stage('2-Run') {
            steps {
                echo "Run app"
                sh "python3 -m flask run --host=0.0.0.0"
            }
        }
    }
}
